[docs]def cast_to_16_bits(array):
    """ 
    Returns an array in uint16 format. Array values below 0 are cast to 0,
    and array values above (2**16) - 1 are cast to (2**16) - 1.
    
    Parameters
    ----------
    array : array-like
        Array to be cast into 16 bits.
    
    Returns
    -------
    out : ndarray, dtype uint16
    """
    array = n.asarray(array)
    array[ array < 0] = 0
    array[ array > (2**16) - 1] = (2**16) - 1
    return array.astype(n.uint16, order = 'C')



[docs]def resize(array, resolution = RESOLUTION):
    """
    Resizes an array to a certain resolution by zero-padding or cropping.
    
    Parameters
    ----------
    array : ndarray
    resolution : 2-tuple
    """
    if array.shape == resolution:
        return array
    elif array.size > resolution[0]*resolution[1]:
        base = n.zeros(shape = resolution, dtype = n.float)
        base[:resolution[0],:resolution[1]] = array
    else:
        base = n.zeros(shape = resolution, dtype = n.float)
        base[:array.shape[0],:array.shape[1]] = array
    return base

    

[docs]def read(filename, normalize = False):
    """ 
    Returns a ndarray from an image filename.  Only TIFF are supported.
    
    Parameters
    ----------
    filename : str
        Absolute path to the file.
    normalize : bool, optional
        If True, normalized image to the total number of counts. Default is False.
    
    Returns
    -------
    image : ndarray, dtype float
        
    Raises
    ------
    ImageNotFoundError
    """
    try:
        image = imread(filename).astype(n.float)
    except:
        raise ImageNotFoundError('Image {} not found.'.format(filename))
    if image.shape != RESOLUTION:
        image = resize(image, RESOLUTION)

    # Normalize to total counts
    if normalize is True:
        image = image/n.sum(image)
    return image
